يتطلع المنتخب الفرنسي إلى إحراز العلامة الكاملة في البطولات التي تستضيفها بلاده والتتويج باللقب الثالث على أرضه فيما يطمح المنتخب البرتغالي بقيادة نجمه رونالدو إلى الفوز بلقبه الأول في البطولات الكبيرة عندما يلتقي نظيره الفرنسي اليوم في المباراة النهائية النارية لبطولة يورو  أهم المسابقات القارية على الإطلاق وسبق للمنتخب الفرنسي أن فاز بكل من البطولتين السابقتين اللتين أقيمتا على أرضه حيث توج بلقب يورو  وكأس العالم  كما أحرز لقب يورو  فيما لا يزال سجل المنتخب البرتغالي خاليا من أي ألقاب فرنسا والبرتغال في نهائي صناعة المجد و الهاتريك الليلة يقف البرتغالي كريستيانو رونالدو وكراته الذهبية الثلاث بين فرنسا وإحرازها لقبا قاريا ثالثا عندما يتواجهان اليوم الأحد في باريس في نهائي كأس أوروبا  لكرة القدم تبدو فرنسا مرشحة قوية لإحراز هاتريك أي اللقب الثالث بعد  و إذ ترشحها مكاتب المراهنات بأرقام مضاعفة عن البرتغال الباحثة عن أول ألقابها الكبرى مع العلم أن الأخيرة خسرت نهائي  بشكل مفاجئ على أرضها أمام اليونان صفر  عندما كان رونالدو في التاسعة عشرة من عمره يأمل منتخب الديوك إحراز لقبه الكبير الثالث على أرضه بعد كأس أوروبا  وكأس العالم  على الملعب عينه وبإمكان فرنسا أن تحلم بوجود جيل جديد يقوده انطوان غريزمان مهاجم أتلتيكو مدريد الإسباني الذي جعل هجوم فرنسا إحدى العلامات الفارقة في البطولة بوجود ديميتري باييت وأوليفييه جيرو والشابين كينغسلي كومان وانطوني مارسيال أيضا كان غريزمان  عاما في قمة التعاسة بعدما أهدر ركلة جزاء في نهائي دوري أبطال أوروبا ضد ريال مدريد الإسباني وكانت بدايته بطيئة في يورو  في المباراة الافتتاحية ضد رومانيا   لكن غريزي فرض نفسه نجما للبطولة بتسجيله ستة أهداف حتى الآن بينها هدفا الفوز على إيرلندا في ثمن النهائي وعلى ألمانيا في نصف النهائي أصبح غريزمان أول لاعب يسجل  أهداف أو أكثر منذ التشيكي ميلان باروش في نسخة   أهداف واللاعب الوحيد الذي يتفوق عليه بعدد الأهداف في نسخة واحدة هو مواطنه بلاتيني  عام  على الطرف المقابل تعول البرتغال بشكل كبير على عملاقها رونالدو لكن فرنسا تريد أن تمنعه من تجاوز رقم ابنها بلاتيني التهديفي بعدما تساويا برصيد  أهداف كأكثر من سجل في تاريخ بطولات أوروبا رونالدو ابن ماديرا الذي بكى كثيرا بعد نهائي  أمام اليونان يبحث عن تسجيل اسمه بأحرف ذهبية في تاريخ المنتخب البرتغالي من خلال تحقيق ما عجز عنه كثيرون من كبار منتخب سيليساو وآخرهم لويس فيغو انتظر رونالدو  عاما و هدفا في  مباراة دولية اللحظة
